In the first part, we imported the cancer and normal counts on each genes. We added a pseudo counts on our observations to avoid dividing by zero, this does not impact our result on a large scale because the log fold of change would be very large if a gene is unregulated/down regulated.

cancer_summary <- cancer_counts |>
  group_by(GeneId) |>
  summarise(cancer_mean_count = mean(count))

normal_summary <- normal_counts |>
  group_by(GeneId) |>
  summarise(normal_mean_count = mean(count))

merged_counts <- bind_rows(normal_counts, cancer_counts)
merged_count_summary <- merged_counts |>
  group_by(GeneId, type) |>
  summarise(mean_count = mean(count), standard_dev_count = sd(count), min_count = min(count), mx_count = max(count))
`summarise()` has grouped output by 'GeneId'. You can override using the
`.groups` argument.

T_test_results <- merged_counts |>
  group_by(GeneId) |>
  group_map(~ t.test(count ~ type, .x))
p_values <- sapply(T_test_results, function(T_test) T_test$p.value)
p_values_data_frame <- data.frame(p_value = p_values)

ggplot(data = p_values_data_frame, aes(x = p_value)) +
  geom_histogram(binwidth = 0.01) +
  labs(
    title = "P values of merged counts",
    x = "p_value"
    )

combine_counts_summary <- cancer_summary |>
  left_join(normal_summary) |>
  mutate(log_fold_of_change = log10(cancer_mean_count / normal_mean_count)) |>
  mutate(corrected_pvals = p.adjust(p_values, method="BH")) |>
  mutate(log_of_pvals = log10(corrected_pvals)) |>
  mutate(selected = log_of_pvals < -10.25 & log_fold_of_change > 2)
